package ua.odu.op.oneforosov;

public class Car {
    String brand;
    String model;
    int year;
    double fuelConsumption;

    public Car() {}

    public Car(int y, String m, String b, double fc) {
        brand = b;
        model = m;
        year = y;
        fuelConsumption = fc;
    }

    public Car(int year) {
        this.year = year;
    }

    public String getInfo() {
        return "Our car is: " + brand + "; Model: " + model + "; Year: " + year;
    }

    public double calculateConsumption(double distance, double fuelPrice) {
        double fuelPer100km = (fuelConsumption / 100) * distance;
        return fuelPer100km * fuelPrice;
    }

    public static void main(String[] args) {
        Car firstCar = new Car( 1997,  "Corolla",  "Toyota", 10.5);
        Car secondCar = new Car( 2002,  "Impreza WRX STI",  "Subaru", 12.1);
        Car thirdCar = new Car( 1999,  "Skyline GT-R",  "Nissan", 13.3);
        Car fourthCar = new Car( 2004,  "RX-8",  "Mazda", 10.8);
        Car fifthCar = new Car( 2007,  "Lancer Evolution IX",  "Mitsubishi", 10.6);
        Car empty = new Car();
        Car single = new Car(1989);
        double fuelPrice = 51.4;
        System.out.println(firstCar.getInfo());
        System.out.println(secondCar.getInfo());
        System.out.println(thirdCar.getInfo());
        System.out.println(fourthCar.getInfo());
        System.out.println(fifthCar.getInfo());
        System.out.println("====================");
        System.out.println("Consumption for firstCar:" + firstCar.calculateConsumption(30, fuelPrice));
        System.out.println("Consumption for secondCar:" + secondCar.calculateConsumption(545.25, fuelPrice));
        System.out.println("Consumption for thirdCar:" + thirdCar.calculateConsumption(133, fuelPrice));
        System.out.println("Consumption for fourthCar:" + fourthCar.calculateConsumption(108, fuelPrice));
        System.out.println("Consumption for fifthCar:" + fifthCar.calculateConsumption(106, fuelPrice));
    }
}
